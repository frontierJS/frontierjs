---
title: completion:generate
description: Print the shell completion script — pipe into your shell config to install
alias: cgen
examples:
  - fli completion:generate
  - fli completion:generate --shell zsh
  - fli completion:generate --shell bash
  - fli completion:generate --shell fish
flags:
  shell:
    char: s
    type: string
    description: "Shell to generate for: bash | zsh | fish (default: auto-detect)"
    defaultValue: ''
---

<script>
const detectShell = () => {
  const shell = process.env.SHELL || ''
  if (shell.includes('zsh'))  return 'zsh'
  if (shell.includes('fish')) return 'fish'
  if (shell.includes('bash')) return 'bash'
  return 'bash'  // safe default
}

// ─── Bash ─────────────────────────────────────────────────────────────────────
// Key technique: we pass COMP_LINE (the raw line up to cursor) to completion:query
// rather than COMP_WORDS. This sidesteps bash's colon-as-word-break issue entirely —
// our query parser splits on whitespace and treats "deploy:logs" as one word.
const bashScript = () => `# fli bash completion
# Generated by: fli completion:generate --shell bash
# Install:  eval "$(fli completion:generate --shell bash)"
# Or:       fli completion:generate --shell bash >> ~/.bash_completion

_fli_completion() {
  local cur line
  cur="\${COMP_WORDS[COMP_CWORD]}"
  # Pass the raw command line up to the cursor — avoids colon word-splitting
  line="\${COMP_LINE:0:\$COMP_POINT}"
  local IFS=$'\\n'
  COMPREPLY=(\$(compgen -W "\$(fli completion:query "\$line" 2>/dev/null)" -- "\$cur"))
}
complete -F _fli_completion fli
`

// ─── Zsh ──────────────────────────────────────────────────────────────────────
// Uses compdef. Zsh doesn't split on colons, so deploy:logs works cleanly.
const zshScript = () => `# fli zsh completion
# Generated by: fli completion:generate --shell zsh
# Install:  eval "$(fli completion:generate --shell zsh)"
# Or:       fli completion:generate --shell zsh > "\${fpath[1]}/_fli"

_fli() {
  local line="\${(j: :)words}"
  local -a completions
  completions=("\${(@f)\$(fli completion:query "\$line" 2>/dev/null)}")
  [[ \${#completions} -eq 0 ]] && return
  compadd -a completions
}
compdef _fli fli
`

// ─── Fish ─────────────────────────────────────────────────────────────────────
// Fish is the simplest — it calls our query with the current commandline.
// Fish doesn't have the colon problem at all.
const fishScript = () => `# fli fish completion
# Generated by: fli completion:generate --shell fish
# Install:  fli completion:generate --shell fish > ~/.config/fish/completions/fli.fish

# Disable default file completions for fli
complete -c fli -f

# Complete commands and flags dynamically
complete -c fli -a "(fli completion:query (commandline -cp) 2>/dev/null)"
`
</script>

```js
const shell = flag.shell || detectShell()

switch (shell) {
  case 'zsh':
    process.stdout.write(zshScript())
    break
  case 'fish':
    process.stdout.write(fishScript())
    break
  case 'bash':
  default:
    process.stdout.write(bashScript())
    break
}
```
